void multiplica(int **d, int **e, int  **f, int anla, int anca, int ancb){
    int i, j,x, p;
    int anlb=anla, anlc=anla, ancc=ancb;
    printf("---------------------------- 1 - Matriz Gerada ---------------------------------\n\n");

    for(i = 0; i < anla; i++) {
        for(j = 0; j < anca; j++) {
            d[i][j]=i+j;
            printf("matriz A =[%d] [%d] = %d |",i,j, d[i][j]);

        }
        printf("\n\n");
    }

    printf("---------------------------- 2 - Matriz Gerada ---------------------------------\n\n");
    for(i = 0; i < anlb; i++) {
        for(j = 0; j < ancb; j++) {
            e[i][j]=i+j;
            printf("matriz A =[%d] [%d] = %d |",i,j, e[i][j]);
        }
        printf("\n\n");
    }

    printf("---------------------------- 3 - Matriz Gerada ---------------------------------\n\n");


    for(i = 0; i < anlc; i++) {
        for(j = 0; j < ancc; j++) {

            x=0;
            for(p=0; p<anca; p++){
                x=x+d[i][p]*e[p][j];
            }
            f[i][j]=x;
            printf("matrizC =[%d] [%d] = %d |",i,j, f[i][j]);
        }
        printf("\n\n");
    }
    free(d[0]);
    free(d);
    free(e[0]);
    free(e);
    free(f[0]);
    free(f);

}

    int main() {

        int i, nla, nca, ncb;
        int **a,**b, **c;

        printf("\n Informe a quntidade de linhas da matriz A : ");
        scanf("%d",&nla);
        printf("\n Informe a quantidade de colunas da matriz A : ");
        scanf("%d",&nca);
        printf("\n Informe a quantidade de colunas da matriz B : ");
        scanf("%d",&ncb);

        a=malloc(nla*sizeof (int*));
        for(i=0; i<nla; i++){
            a[i]=malloc(nca*sizeof (int));
        }

        b=malloc(nca*sizeof (int*));
        for(i=0; i<nca; i++){
            b[i]=malloc(ncb*sizeof (int));
        }

        c=malloc(nla*sizeof (int*));
        for(i=0; i<nla; i++){
            c[i]=malloc(ncb*sizeof (int));
        }


        (*multiplica)(a, b, c, nla, nca, ncb);
        free(a[0]);
