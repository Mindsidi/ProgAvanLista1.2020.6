#include <stdio.h>
#include <stdlib.h>
struct nomedovetor1{
    char nome[20];
};
struct nomedovetor2{
    char nome[20];
};
struct nomedovetorsoma{
    char nome[20];
};
int main()
{
    int *x, *y, *z, i;
    int  nz;
    struct nomedovetor1 a;
    struct nomedovetor2 b;
    struct nomedovetorsoma c;
    printf("Nome do vetor 1=");
    fgets(a.nome, 20, stdin);
    printf("Nome do vetor 2=");
    fgets(b.nome, 20, stdin);
    printf("Nome do vetor soma=");
    fgets(c.nome, 20, stdin);
    printf("tamanho do vetor \n");
    scanf("%d", &nz);
    x=malloc(nz *sizeof (int));
    y=malloc(nz *sizeof (int));
    z=malloc(nz *sizeof (int));

    for(i=0; i<nz; i++){
        x[i]=0;
    }
    for(i=0; i<nz; i++){
        y[i]=0;
    }
    for(i=0; i<nz; i++){
        x[i]=rand()%20 -10;
    }
    printf("O valor do ");
    printf("%s\n", a.nome);
    for(i=0; i<nz; i++){
        printf("%d,", x[i]);
    }
    printf("\n");
    for(i=0; i<nz; i++){
        y[i]=rand()%20 -10;
    }
    printf("O valor do ");
    printf("%s\n", b.nome);
    for(i=0; i<nz; i++){
        printf("%d,", y[i]);
    }
    for(i=0; i<nz; i++){
        z[i]=0;
    }
      printf("\n");
    for(i=0; i<nz; i++){
        z[i]=x[i]+y[i];
    }
    printf("O valor do ");
    printf("%s\n", c.nome);
    for(i=0; i<nz; i++){
        printf("%d,", z[i]);
    }
    printf("\n");
    return 0;
}
