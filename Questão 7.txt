	p = mat + 1;    //Legal
	p = mat++;      //Ilegal, não pode incrementar valor do tipo int ao ponteiro.
	p = ++mat;      //Ilegal, não pode incrementar valor do tipo int ao ponteiro.
	x = (*mat)++;   //Ilegal, não pode incrementar valor do tipo int ao ponteiro.
