void soma (int a, int b){
    printf("Soma=%d\n", a+b);
}

}
void multiplicacao(int a, int b){
    printf("multiplicacao=%d\n", a*b);
}

{
    void (*pf_array[])(int, int)={soma, subtracao, multiplicacao};
   int opcao, a=3, b=4;
